module ind_gen(
clock,
k,
ready,
reset
out[13:0]
);

input clock;
input k;
input ready;
input reset;
output out[13:0]; 


reg cnt[13:0];
assign out = reg;

//use counter to get ind
always @ (negedge ready or posedge reset or posedge clock)
begin
	if(!ready | reset) begin
		cnt <= 13'd0;
	end

	else if(cnt==1055 & !k) begin
		cnt <= cnt;
	end
	else if(cnt==6143) begin 
		cnt <= cnt;
	end 
	else begin cnt <= cnt+1;
	end
end

endmodle
